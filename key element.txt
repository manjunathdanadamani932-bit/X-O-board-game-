#include <stdio.h>
int main()
{
    int n, i, key, flag = 0;

    printf("Enter the size of array: ");
    scanf("%d", &n); 

    int arr[n];
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) 
    {
        scanf("%d", &arr[i]);
    }
    printf("Enter the key : ");
    scanf("%d", &key);
    for (i = 0; i < n; i++) {
        if (arr[i] == key) {
            flag = 1;
            break;
        }
    }
    if (flag == 1) {
        printf("The Key is found");
    } else {
        printf("The Key is  not found");
    }

    return 0;
}
